def undeploy():
    """ API function to undeploy the application with a specific id """
    id_app = request.form["id_app"]
    response = submitter.undeploy(id_app)
    if response is None:
        return "<h1>correctly undeployed</h1>\n"
    else:
        return "<h1>{}</h1>\n".format(response)
